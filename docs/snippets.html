2.1 Extract dob and gender from chi

library(tidyverse)
library(lubridate)
x = tibble(
  chi = c("1009761234", "1009762345")
  )

x %>%
  mutate(
    dob = str_extract(chi, ".{6}") %>% 
      dmy(),
    gender = str_sub(chi, 9, 9) %>% 
      as.numeric() %>% 
      {ifelse(. %% 2 == 0, "Female", "Male")}
  )
## # A tibble: 2 x 3
##   chi        dob        gender
##   <chr>      <date>     <chr> 
## 1 1009761234 1976-09-10 Male  
## 2 1009762345 1976-09-10 Female
